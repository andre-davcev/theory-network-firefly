cities (service)
    public const earthRadius: number = 6371;

    public static degrees2Radians(degrees: number): number
    {
        return degrees * (Math.PI / 180);
    }

    public static distanceBetweenPoints(geopoint1: firestore.GeoPoint, geopoint2: firestore.GeoPoint)
    {
        return ServiceCity.distanceBetween(geopoint1.latitude, geopoint1.longitude, geopoint2.latitude, geopoint2.longitude);
    }

    // Haversine Formula (km)
    public static distanceBetween(latitude1, longitude1, latitude2, longitude2)
    {
        const distanceLatitude:  number = degrees2Radians(latitude2  - latitude1);
        const distanceLongitude: number = degrees2Radians(longitude2 - longitude1);
        const a: number = Math.sin(distanceLatitude/2) * Math.sin(distanceLatitude/2) +
                          Math.cos(degrees2Radians(latitude1)) * Math.cos(degrees2Radians(latitude2)) *
                          Math.sin(distanceLongitude/2) * Math.sin(distanceLongitude/2);
        const c: number = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

        return ServiceCity.earthRadius * c; // Distance in km
    }
