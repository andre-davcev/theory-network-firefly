{
    version     : string;
    id          : string;
    userId      : string;
    dateCreated : firestore.FieldValue;
    dateUpdated : firestore.FieldValue;
    metadata?   : Record<string, any>;
}

    function documentValidCreate(uid, id, data)
    {
        return data.version     == null &&
               data.id          == null &&
               data.userId      == uid  &&
               data.dateCreated == null &&
               data.dateUpdated == null &&
               data.metadata    == null;
    }

    function documentValidUpdate(uid, dataOld, dataNew)
    {
        return dataNew.version     == dataOld.version     &&
               dataNew.id          == dataOld.id          &&
               dataNew.userId      == dateOld.userId      &&
               dataNew.dateCreated == dataOld.dateCreated &&
               dataNew.dateUpdated is timestamp           &&
               dataNew.metadata    == null;
    }

    }

    function eventValidCreate(uid, id, data)
    {
        return documentValidCreate(uid, id, data) &&
               data.name            is string     &&
               data.tagline         is string     &&
               data.description     is string     &&
               data.bucketPath      is string     &&
               data.private         is boolean    &&
               data.geopoint        is geopoint   &&
               data.city            != null       &&
               data.city.geopoint   is geopoint   &&

               data.clusters        is array      &&
               data.clusters.size() == 1          &&
               !data.notifyComplete;
    }

    function eventValidUpdate(uid, dataOld, dataNew)
    {
        return documentValidUpdate(uid, dataOld, dataNew)         &&
               dataNew.name            is string                  &&
               dataNew.tagline         is string                  &&
               dataNew.description     is string                  &&
               dataNew.bucketPath      is string                  &&
               dataNew.private         is boolean                 &&
               dataNew.geopoint        is geopoint                &&
               dataNew.city            != null                    &&
               dataNew.city.geopoint   is geopoint                &&
               dataNew.city.cityId     is string                  &&
               dataNew.city.city       is string                  &&
               dataNew.city.region     is string                  &&
               dataNew.city.country    is string                  &&
               dataNew.timeStart       is string                  &&
               dataNew.timeEnd         is string                  &&
               dataNew.timeNotify      is string                  &&
               dataNew.clusters        is array                   &&
               dataNew.clusters.size() == dataOld.clusters.size() &&
               !data.notifyComplete;
    }
